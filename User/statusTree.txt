status: 全局状态树根节点
      state: 小车运行状态与控制参数
            T: 系统周期，单位 ms
            time: 系统累计运行时间
            motion: 当前运动模式，STOP/KEEP_ANGLE/FIND_LINE
            initial_angle: 初始参考角
            cur_angle: 当前角度
            tar_angle: 目标角度
            base_speed: 基础速度
            road_determine: 路口识别状态
                  data_buf: 当前一帧路口原始数据
                  integral: 多帧累计结果
                  maybe: 候选路口计数器
                  cross_cnt: 路口计数
                  cross: 当前识别到的道路类型
                  integral_times: 判路累计帧数
            gw_8bit: 预留的8位灰度状态值
            status_pid: 状态层PID集合
                  follow_line_pid: 巡线PID
                        kp: 比例系数
                        ki: 积分系数
                        kd: 微分系数
                        T: PID采样周期
                        error: 当前误差
                        last_error: 上次误差
                        integral: 当前积分项
                        derivative: 当前微分项
                        integral_max: 积分限幅
                        is_first: 是否首次计算
                  keep_angle_pid: 保角PID
                        kp: 比例系数
                        ki: 积分系数
                        kd: 微分系数
                        T: PID采样周期
                        error: 当前误差
                        last_error: 上次误差
                        integral: 当前积分项
                        derivative: 当前微分项
                        integral_max: 积分限幅
                        is_first: 是否首次计算
      sensor: 传感器数据集合
            gy901: 陀螺仪状态
                  data_buf[24]: 原始数据缓冲区
                  device_addr: I2C设备地址
                  data_start_addr: 数据起始寄存器地址
                  gy901_keep_angle_pid: 陀螺仪自带保角PID参数
                        kp: 比例系数
                        ki: 积分系数
                        kd: 微分系数
                        T: PID采样周期
                        error: 当前误差
                        last_error: 上次误差
                        integral: 当前积分项
                        derivative: 当前微分项
                        integral_max: 积分限幅
                        is_first: 是否首次计算
            gw_8bit: 8路数字灰度传感器状态
                  data_buf: 8位灰度原始位图
                  gw_bit_weight[8]: 各通道权重
                  integral: 多帧累计结果
                  maybe: 候选路口计数器
                  cross_cnt: 路口计数
                  cross: 当前识别到的道路类型
                  gw_find_line_pid: 数字灰度巡线PID
                        kp: 比例系数
                        ki: 积分系数
                        kd: 微分系数
                        T: PID采样周期
                        error: 当前误差
                        last_error: 上次误差
                        integral: 当前积分项
                        derivative: 当前微分项
                        integral_max: 积分限幅
                        is_first: 是否首次计算
                  gw_diff: 计算得到的偏差
            gw_analogue: 8路模拟灰度传感器状态
                  channel[8]: 8路原始采样值
                  sta: 工作/校准状态
                  correction_data_w[8]: 白底校准数据
                  correction_data_b[8]: 黑底校准数据
                  digital_8bit: 二值化后的8位数据
                  digital_high_threshold[8]: 高阈值
                  digital_low_threshold[8]: 低阈值
                  diff: 模拟巡线偏差
      motor: 电机执行器状态
            wheel[4]: 四个轮电机状态数组
                  which: 电机编号
                  trust: 当前输出/推力值
                  cur_speed: 当前速度
                  tar_speed: 目标速度
                  dir: 电机正方向定义
                  wheel_pid: 单轮速度PID
                        kp: 比例系数
                        ki: 积分系数
                        kd: 微分系数
                        T: PID采样周期
                        error: 当前误差
                        last_error: 上次误差
                        integral: 当前积分项
                        derivative: 当前微分项
                        integral_max: 积分限幅
                        is_first: 是否首次计算
            servo[2]: 两个舵机状态数组
                  which: 舵机编号
                  angle: 当前目标角度
                  max_angle: 舵机最大角度范围
      device: 板载与外设状态
            led_on_board: 板载LED状态
                  which: 设备编号
                  High_level_is_on: 高电平是否点亮
                  on: 当前开关状态
            led1: 外接LED1状态
                  which: 设备编号
                  High_level_is_on: 高电平是否点亮
                  on: 当前开关状态
            led2: 外接LED2状态
                  which: 设备编号
                  High_level_is_on: 高电平是否点亮
                  on: 当前开关状态
            button_D2: 按键D2状态
                  last: 上一次电平状态
                  now: 当前电平状态
                  Press_is_high_level: 按下是否为高电平
                  which: 按键编号
                  long_press_cnt: 长按计数器
            button_B11: 按键B11状态
                  last: 上一次电平状态
                  now: 当前电平状态
                  Press_is_high_level: 按下是否为高电平
                  which: 按键编号
                  long_press_cnt: 长按计数器
            buzzer: 蜂鸣器状态
                  which: 设备编号
                  High_level_is_on: 高电平是否鸣叫
                  on: 当前开关状态
